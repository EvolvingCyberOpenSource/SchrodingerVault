// Ask Rust for all table schemas
const schemas = await window.__TAURI__.core.invoke("debug_list_schema");

schemas.forEach(t => {
  console.group(`TABLE: ${t.name}`);
  console.log("SQL:", t.sql);
  console.table(t.columns);
  console.groupEnd();
});
